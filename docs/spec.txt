.NET!

client:
	TcpClient->SslStream tolerating any cert
	
	send: <from> <to> <msg> <file path> -> server -> ticket
	
	recv: <ticket> -> msg info -> file contents
	
server:
	TcpListener->SslStream w/generated-self-signed cert
	
	send:
		creates: ticket: Guid + UtcNow.Ticks + Message Info, all SHA-256'd
		creates msg: from, to, msg, file info, and ticket
		create temp directory with ticket name
		receives file into temp file
		logs send

	recv:
		takes ticket
		looks up message
		sends message info
		logs recv
		waits on client go-ahead
		opens file and delivers file bytes
		deletes message
